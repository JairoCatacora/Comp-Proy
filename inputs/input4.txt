struct Person {
    int age;
    char name[10];
};

int getAge(Person p) {
    return p.age;
}

int main() {
    Person person;
    int a;
    person.age = 25;
    person.name[0] = 'J';
    person.name[1] = 'o';
    person.name[2] = 'h';
    person.name[3] = 'n';
    person.name[4] = 0;
    a = getAge(person);
    print(a);
    person.age = person.age + 1;
    print(person.age);
    return 0;
}
