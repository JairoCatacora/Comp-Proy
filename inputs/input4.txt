#include <stdio.h>

int sumArray(int a[], int n) {
    int i;
    int s;
    i = 0;
    s = 0;
    while (i < n) {
        s = s + a[i];
        i = i + 1;
    }
    return s;
}

int main() {
    int nums[5];
    int total;
    nums[0] = 10;
    nums[1] = 20;
    nums[2] = 30;
    nums[3] = 40;
    nums[4] = 50;
    total = sumArray(nums, 5);
    print(total);
    return 0;
}
