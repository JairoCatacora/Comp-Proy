int suma(int a,int b) {
    return a + b;
}

int factorial(int n) {
    int resultado;
    int i;
    resultado = 1;
    i = 0;
    while (i < n) {
        i = i + 1;
        resultado = resultado * i;
    }
    return resultado;
}

int main() {
    int x;
    int y;
    int resultado;
    x = 5;
    y = 3;
    resultado = suma(x, y);
    print(resultado);
    print(factorial(5));
    return 0;
}
