int suma(int a,int b) {
    return a + b;
}

int factorial(int n) {
    if (n == 0) {
        return 1;
    }
    return n * factorial(n - 1);
}

int main() {
    int x;
    int y;
    int resultado;
    x = 5;
    y = 3;
    resultado = suma(x, y);
    print(resultado);
    print(factorial(5));
    return 0;
}
