#include <stdio.h>

int countTo(int n) {
    int i;
    int sum;
    i = 0;
    sum = 0;
    while (i < n) {
        sum = sum + i;
        i = i + 1;
    }
    return sum;
}

int main() {
    char letters[] = {'A','B','C','D','E'};
    int result;
    result = countTo(10);
    print(result);
    return 0;
}
