int findMax(int arr[], int n) {
    int max;
    int i;
    max = arr[0];
    i = 1;
    while (i < n) {
        if (arr[i] > max) {
            max = arr[i];
        } else {
            max = max;
        }
        i = i + 1;
    }
    return max;
}

int main() {
    int numbers[4];
    int maximum;
    numbers[0] = 15;
    numbers[1] = 42;
    numbers[2] = 8;
    numbers[3] = 23;
    maximum = findMax(numbers, 4);
    print(maximum);
    return 0;
}
