int countTo(int n) {
    int i;
    int sum;
    i = 0;
    sum = 0;
    while (i < n) {
        sum = sum + i;
        i = i + 1;
    }
    return sum;
}

int main() {
    char letters[5];
    int result;
    letters[0] = 'A';
    letters[1] = 'B';
    letters[2] = 'C';
    letters[3] = 'D';
    letters[4] = 'E';
    result = countTo(10);
    print(result);
    return 0;
}
